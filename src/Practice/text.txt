import java.util.Scanner;
class Program{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        pattern3(n);
    }

    /*
    A
    B B
    C C C
    D D D D
    E E E E E

    */
    static void pattern3(int n){
        for(int i = 1; i<=n;i++){
            for(int j = 1; j<=i;j++){
                System.out.print((char)(i+96));
            }
            System.out.println();
        }
    }
}
